#### If statement
```python
    x = 10 
    if x > 5 
    {
        print("x is greater than 5");
    }
    elif x < 5
    {
        print("x is less than 5")
    }
    else
    {
        print("x equals to 5")
    }

    if true or false
    {
        print("true")
    }
    
    if true and !false
    {
        print("true")
    }
```

#### For loop   
```python

    }
```

#### Program snippet
```python
    
PROGRAM Part10
{
    function fib(n: int) {
        if n < 1 {
            return 0;
        }
        elif n < 3 {
            return 1;
        }
        return fib(n - 1) + fib(n - 2);
    }


    for i = 0; i < 10; i = i + 1 {
        print(fib(i));
    }

}
```
